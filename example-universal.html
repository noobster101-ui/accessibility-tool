Angular Angular 14+

Lazy load React component in Angular application.

1. Install Dependencies

npm install react react-dom react-accessibility-tool
# For Angular 17+ with standalone components
npm install @angular/core@latest @angular/common@latest

2. Create React Component (src/app/accessibility-tool.component.tsx)

import { Component, OnInit, OnDestroy } from '@angular/core';
import * as React from 'react';
import * as ReactDOM from 'react-dom/client';
import AccessibilityTool from 'react-accessibility-tool';

@Component({
  selector: 'app-accessibility-tool',
  template: '
',
  standalone: true,
})
export class AccessibilityToolComponent implements OnInit, OnDestroy {
  private container: HTMLElement | null = null;
  private root: any = null;

  ngOnInit() {
    // Create container element
    this.container = document.createElement('div');
    this.container.id = 'accessibility-react-wrapper';
    this.container.style.position = 'fixed';
    this.container.style.zIndex = '99999';
    document.body.appendChild(this.container);

    // Mount React component
    this.root = ReactDOM.createRoot(this.container);
    this.root.render(React.createElement(AccessibilityTool));
  }

  ngOnDestroy() {
    if (this.root) {
      this.root.unmount();
    }
    if (this.container) {
      document.body.removeChild(this.container);
    }
  }
}

License API Response Structure

// LicenseResult object structure:
{
  valid: boolean,           // Is license valid
  authorized: boolean,      // Is user authorized
  tier: 'free'|'pro'|'enterprise',
  domain: string,           // Licensed domain
  authorizedAt: string,     // ISO date string
  expiresAt: Date|null,     // Expiration date
  message: string,          // Status message
  fromCache: boolean,       // Came from cache
  error: string|null        // Error message if failed
}
